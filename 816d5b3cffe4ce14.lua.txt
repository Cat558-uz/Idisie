local function createButton(text, color, pos)
	local Btn = Instance.new("TextButton", Frame)
	Btn.Size = UDim2.new(0.4, 0, 0.35, 0)
	Btn.Position = pos
	Btn.BackgroundColor3 = color
	Btn.Text = text
	Btn.TextColor3 = Color3.new(1,1,1)
	Btn.Font = Enum.Font.SourceSansBold
	Btn.TextSize = 18
	Btn.BorderSizePixel = 0
	Instance.new("UICorner", Btn).CornerRadius = UDim.new(0, 8)

	local hover = TweenService:Create(Btn, TweenInfo.new(0.2), {BackgroundColor3 = color:Lerp(Color3.new(1,1,1),0.2)})
	local unhover = TweenService:Create(Btn, TweenInfo.new(0.2), {BackgroundColor3 = color})
	local clickAnim = TweenService:Create(Btn, TweenInfo.new(0.1), {Size = UDim2.new(0.38, 0, 0.33, 0)})

	Btn.MouseEnter:Connect(function() hover:Play() end)
	Btn.MouseLeave:Connect(function() unhover:Play() end)
	Btn.MouseButton1Down:Connect(function() clickAnim:Play() end)
	Btn.MouseButton1Up:Connect(function()
		unhover:Play()
		TweenService:Create(Btn, TweenInfo.new(0.1), {Size = UDim2.new(0.4, 0, 0.35, 0)}):Play()
	end)

	return Btn
end

-- Arrastar
local dragging, dragInput, dragStart, startPos
main.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = main.Position
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

main.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end
end)

game:GetService("UserInputService").InputChanged:Connect(function(input)
	if input == dragInput and dragging then
		local delta = input.Position - dragStart
		main.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
end)
